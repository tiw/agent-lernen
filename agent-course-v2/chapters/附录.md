## B. 常见 LLM API 对接指南

### 1. OpenAI API

```python
"""
OpenAI API 对接

安装: pip install openai
文档: https://platform.openai.com/docs
"""

from __future__ import annotations

import os
from openai import AsyncOpenAI


class OpenAIClient:
    """OpenAI 异步客户端"""

    def __init__(
        self,
        api_key: str | None = None,
        model: str = "gpt-4o",
        base_url: str | None = None,
    ):
        self.model = model
        self.client = AsyncOpenAI(
            api_key=api_key or os.getenv("OPENAI_API_KEY"),
            base_url=base_url,  # 兼容兼容 OpenAI 接口的服务
        )

    async def chat(
        self,
        prompt: str,
        system_prompt: str = "你是一个有用的助手。",
        temperature: float = 0.7,
        max_tokens: int = 4096,
    ) -> str:
        """简单对话"""
        response = await self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": prompt},
            ],
            temperature=temperature,
            max_tokens=max_tokens,
        )
        return response.choices[0].message.content

    async def stream_chat(
        self,
        prompt: str,
        system_prompt: str = "你是一个有用的助手。",
        temperature: float = 0.7,
    ):
        """流式对话"""
        stream = await self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": prompt},
            ],
            temperature=temperature,
            stream=True,
        )
        async for chunk in stream:
            delta = chunk.choices[0].delta
            if delta.content:
                yield delta.content
```

```python
安装: pip install anthropic
文档: https://docs.anthropic.com

from __future__ import annotations

import os
from anthropic import AsyncAnthropic


class AnthropicClient:
    """Anthropic 异步客户端"""

    def __init__(
        self,
        api_key: str | None = None,
        model: str = "claude-3-5-sonnet-20241022",
        base_url: str | None = None,
    ):
        self.model = model
        self.client = AsyncAnthropic(
            api_key=api_key or os.getenv("ANTHROPIC_API_KEY"),
            base_url=base_url,
        )

    async def chat(
        self,
        prompt: str,
        system_prompt: str = "你是一个有用的助手。",
        temperature: float = 0.7,
        max_tokens: int = 4096,
    ) -> str:
        """简单对话"""
        response = await self.client.messages.create(
            model=self.model,
            system=system_prompt,
            messages=[{"role": "user", "content": prompt}],
            temperature=temperature,
            max_tokens=max_tokens,
        )
        return response.content[0].text

    async def stream_chat(
        self,
        prompt: str,
        system_prompt: str = "你是一个有用的助手。",
        temperature: float = 0.7,
        max_tokens: int = 4096,
    ):
        """流式对话"""
        async with self.client.messages.stream(
            model=self.model,
            system=system_prompt,
            messages=[{"role": "user", "content": prompt}],
            temperature=temperature,
            max_tokens=max_tokens,
        ) as stream:
            async for text in stream.text_stream:
                yield text
```

### 3. 本地模型（Ollama）

```python
"""
本地模型对接 —— Ollama

安装:
  1. 安装 Ollama: https://ollama.com
  2. 拉取模型: ollama pull qwen2.5:7b
  3. 启动服务: ollama serve

安装: pip install ollama
"""

from __future__ import annotations

import ollama


class OllamaClient:
    """Ollama 客户端"""

    def __init__(self, model: str = "qwen2.5:7b", host: str = "http://localhost:11434"):
        self.model = model
        self.client = ollama.AsyncClient(host=host)

    async def chat(
        self,
        prompt: str,
        system_prompt: str = "你是一个有用的助手。",
        temperature: float = 0.7,
    ) -> str:
        response = await self.client.chat(
            model=self.model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": prompt},
            ],
            options={"temperature": temperature},
        )
        return response["message"]["content"]

    async def stream_chat(
        self,
        prompt: str,
        system_prompt: str = "你是一个有用的助手。",
        temperature: float = 0.7,
    ):
        response = await self.client.chat(
            model=self.model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": prompt},
            ],
            options={"temperature": temperature},
            stream=True,
        )
        async for chunk in response:
            if chunk["message"]["content"]:
                yield chunk["message"]["content"]
```

### 4. 统一客户端接口

```python
"""
统一 LLM 客户端接口

支持多种后端，统一 API 风格
"""

from __future__ import annotations

from enum import Enum


class Provider(str, Enum):
    OPENAI = "openai"
    ANTHROPIC = "anthropic"
    OLLAMA = "ollama"
    CUSTOM = "custom"


def create_client(
    provider: Provider,
    model: str,
    api_key: str | None = None,
    base_url: str | None = None,
) -> object:
    """创建统一的 LLM 客户端"""
    if provider == Provider.OPENAI:
        from openai import AsyncOpenAI
        return AsyncOpenAI(api_key=api_key, base_url=base_url)
    elif provider == Provider.ANTHROPIC:
        from anthropic import AsyncAnthropic
        return AsyncAnthropic(api_key=api_key, base_url=base_url)
    elif provider == Provider.OLLAMA:
        import ollama
        return ollama.AsyncClient(host=base_url or "http://localhost:11434")
    else:
        raise ValueError(f"不支持的 provider: {provider}")
```

## C. 性能优化清单

### LLM 调用优化

| 优化项 | 方法 | 效果 |
|--------|------|------|
| 减少 prompt 长度 | 只发送必要的上下文，使用摘要替代完整历史 | 降低 30-50% token 消耗 |
| 缓存响应 | 对相同的 prompt 缓存结果 | 减少重复调用 |
| 批量请求 | 合并多个小请求为一个 | 减少 API 调用次数 |
| 选择合适模型 | 简单任务用小模型，复杂任务用大模型 | 降低成本 60-80% |
| 流式输出 | 使用 stream=True，边生成边显示 | 降低感知延迟 |
| 设置 max_tokens | 限制最大输出长度 | 防止过度生成 |
| 调整 temperature | 确定性任务用 0，创意任务用 0.7 | 控制输出质量 |

### 代码执行优化

| 优化项 | 方法 | 效果 |
|--------|------|------|
| 命令超时 | 设置合理的超时时间（30-60 秒） | 防止卡死 |
| 并行执行 | 独立的工具调用并行执行 | 减少总耗时 |
| 结果缓存 | 缓存文件读取、目录列表等结果 | 减少重复 I/O |
| 增量处理 | 只处理变化的文件，而非全量 | 减少处理时间 |
| 连接池 | 复用 HTTP 连接 | 减少连接开销 |

### 内存优化

| 优化项 | 方法 | 效果 |
|--------|------|------|
| 上下文窗口管理 | 定期压缩对话历史 | 防止超出 token 限制 |
| 流式处理大文件 | 分块读取，不一次性加载 | 降低内存峰值 |
| 及时释放资源 | 使用完毕后关闭连接和文件句柄 | 防止内存泄漏 |
| 限制缓存大小 | 设置 LRU 缓存上限 | 控制内存使用 |

### 具体代码示例

```python
from __future__ import annotations

# ❌ 不好的做法：发送完整历史
messages = full_conversation_history  # 可能超过 100KB
response = await client.chat(messages)

# ✅ 好的做法：只发送最近的对话 + 摘要
recent = conversation_history[-10:]  # 最近 10 轮

    tool_b.run(),
    tool_c.run(),
```

## D. 故障排查手册

### 1. API 调用失败

| 错误 | 原因 | 解决方案 |
|------|------|---------|
| `401 Unauthorized` | API Key 无效或过期 | 检查环境变量，重新生成 Key |
| `429 Too Many Requests` | 触发速率限制 | 增加重试逻辑，降低请求频率 |
| `500 Internal Server Error` | 服务端错误 | 等待后重试，检查服务状态页 |
| `Connection refused` | 网络不通 | 检查代理设置、防火墙规则 |
| `Timeout` | 响应超时 | 增加 timeout，检查网络质量 |

```python
from __future__ import annotations

# 带重试的 API 调用
import asyncio
from tenacity import retry, stop_after_attempt, wait_exponential

@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=1, min=2, max=30),
    reraise=True,
)
async def chat_with_retry(client, prompt: str) -> str:
    return await client.chat(prompt)
```

### 2. Agent 行为异常

| 现象 | 可能原因 | 排查方法 |
|------|---------|---------|
| Agent 不执行命令 | 安全策略太严格 | 检查 whitelist 配置 |
| Agent 执行了危险命令 | 白名单规则有漏洞 | 审查安全规则，添加测试 |
| Agent 回复乱码 | 编码问题 | 检查文件读取的 encoding 参数 |
| Agent 无限循环 | prompt 设计有问题 | 检查 system prompt，添加最大轮数限制 |
| Agent 忽略用户指令 | 上下文过长 | 压缩历史，确保关键指令在最近 |

### 3. 终端界面问题

| 现象 | 可能原因 | 解决方案 |
|------|---------|---------|
| 输出闪烁 | 频繁刷新 | 降低 refresh_per_second |
| 中文乱码 | 终端编码不支持 | 设置 `export LANG=zh_CN.UTF-8` |
| 补全不工作 | prompt_toolkit 版本问题 | `pip install -U prompt_toolkit` |
| 颜色不显示 | 终端不支持 ANSI | 检查 TERM 环境变量 |

### 4. 文件系统问题

| 现象 | 可能原因 | 解决方案 |
|------|---------|---------|
| 沙箱拒绝合法路径 | allowed_roots 配置错误 | 检查沙箱配置 |
| 符号链接被拒绝 | allow_symlinks=False | 根据需要调整配置 |
| 文件过大被拒绝 | max_file_size 太小 | 调整限制值 |
| 权限错误 | 文件权限不足 | `chmod` 或使用 `sudo` |

## G. 下一步学习方向

完成本教程后，你可以继续探索：

### 🛠️ 深入技术

1. **Fine-tuning**：微调专属模型，让 Agent 更懂你的领域
2. **RAG 进阶**：混合检索、重排序、多跳推理
3. **Agent 评估**：建立评估体系，量化 Agent 能力
4. **多模态**：让 Agent 能看图片、听语音

### 🚀 实战方向

1. **个人助手**：邮件管理、日程安排、信息整理
2. **开发助手**：代码审查、Bug 修复、文档生成
3. **数据分析**：自动报表、异常检测、趋势预测
4. **内容创作**：文章写作、视频脚本、社交媒体

### 🏗️ 架构方向

1. **微服务化**：将 Agent 拆分为独立服务
2. **消息队列**：使用 RabbitMQ/Kafka 处理异步任务
3. **可观测性**：集成 Prometheus + Grafana 监控
4. **CI/CD**：自动化测试和部署流程
