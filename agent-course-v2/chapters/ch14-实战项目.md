### 代码 1：`src/config.py` —— 集中配置管理

```python
from __future__ import annotations

"""
配置管理 —— 集中管理所有 Agent 配置

支持三种配置来源（优先级从高到低）：
1. 环境变量（适合密钥和部署配置）
2. 配置文件（settings.yaml）
3. 默认值
"""

import os
import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

logger = logging.getLogger(__name__)


@dataclass
class LLMConfig:
    """LLM 客户端配置"""
    provider: str = "anthropic"       # anthropic / openai / ollama
    model: str = "claude-3-5-sonnet-20241022"
    api_key: str = ""
    base_url: str = ""
    temperature: float = 0.7
    max_tokens: int = 4096
    timeout: float = 60.0


@dataclass
class SecurityConfig:
    """安全配置"""
    strict_mode: bool = False
    sandbox_enabled: bool = True
    sandbox_root: str = ""            # 空 = 当前目录
    audit_enabled: bool = True
    audit_dir: str = ""               # 空 = ~/.my_agent/audit


@dataclass
class MemoryConfig:
    """记忆系统配置"""
    short_term_max_turns: int = 20    # 短期记忆最大轮数
    long_term_enabled: bool = True
    long_term_dir: str = ""           # 空 = ~/.my_agent/memory
    embedding_model: str = "text-embedding-3-small"


@dataclass
class HookConfig:
    """Hook 系统配置"""
    enabled: bool = True
    config_file: str = ""             # Hook 配置文件路径


@dataclass
class AgentConfig:
    """
    Agent 总配置

    所有子配置的组合，提供统一的配置入口。
    """
    # 基本信息
    name: str = "my_agent"
    version: str = "1.0.0"

    # 子配置
    llm: LLMConfig = field(default_factory=LLMConfig)
    security: SecurityConfig = field(default_factory=SecurityConfig)
    memory: MemoryConfig = field(default_factory=MemoryConfig)
    hooks: HookConfig = field(default_factory=HookConfig)

    # 运行时
    log_level: str = "INFO"
    data_dir: str = ""                # 数据目录，空 = ~/.my_agent

    @classmethod
    def from_env(cls) -> "AgentConfig":
        """
        从环境变量加载配置

        环境变量优先级最高，适合部署时覆盖配置。
        """
        config = cls()

        # LLM 配置
        config.llm.provider = os.getenv("LLM_PROVIDER", config.llm.provider)
        config.llm.model = os.getenv("LLM_MODEL", config.llm.model)
        config.llm.api_key = os.getenv("ANTHROPIC_API_KEY", "") or os.getenv("OPENAI_API_KEY", "")
        config.llm.base_url = os.getenv("LLM_BASE_URL", "")
        config.llm.temperature = float(os.getenv("LLM_TEMPERATURE", config.llm.temperature))
        config.llm.max_tokens = int(os.getenv("LLM_MAX_TOKENS", config.llm.max_tokens))

        # 安全配置
        config.security.strict_mode = (
            os.getenv("STRICT_MODE", "false").lower() == "true"
        )
        config.security.sandbox_enabled = (
            os.getenv("SANDBOX_ENABLED", "true").lower() == "true"
        )
        config.security.sandbox_root = os.getenv("SANDBOX_ROOT", "")
        config.security.audit_enabled = (
            os.getenv("AUDIT_ENABLED", "true").lower() == "true"
        )

        # 记忆配置
        config.memory.short_term_max_turns = int(
            os.getenv("SHORT_TERM_MAX_TURNS", config.memory.short_term_max_turns)
        )

        # 日志
        config.log_level = os.getenv("LOG_LEVEL", config.log_level)

        # 数据目录
        config.data_dir = os.getenv("DATA_DIR", "")

        return config

    @property
    def resolved_data_dir(self) -> Path:
        """获取解析后的数据目录"""
        if self.data_dir:
            return Path(self.data_dir)
        return Path.home() / ".my_agent"

    def to_dict(self) -> dict[str, Any]:
        """转换为字典（用于序列化）"""
        return {
            "name": self.name,
            "version": self.version,
            "llm": {
                "provider": self.llm.provider,
                "model": self.llm.model,
                "temperature": self.llm.temperature,
                "max_tokens": self.llm.max_tokens,
            },
            "security": {
                "strict_mode": self.security.strict_mode,
                "sandbox_enabled": self.security.sandbox_enabled,
            },
            "memory": {
                "short_term_max_turns": self.memory.short_term_max_turns,
                "long_term_enabled": self.memory.long_term_enabled,
            },
            "hooks": {
                "enabled": self.hooks.enabled,
            },
            "log_level": self.log_level,
        }

    def validate(self) -> list[str]:
        """
        验证配置有效性

        返回错误列表，空列表表示配置有效。
        """
        errors = []

        if not self.llm.api_key:
            errors.append("LLM API Key 未设置，请设置 ANTHROPIC_API_KEY 或 OPENAI_API_KEY")

        if self.llm.temperature < 0 or self.llm.temperature > 2:
            errors.append(f"temperature 应在 0-2 之间，当前: {self.llm.temperature}")

        if self.llm.max_tokens < 1:
            errors.append(f"max_tokens 应大于 0，当前: {self.llm.max_tokens}")

        return errors
```

### 代码 2：`src/core/agent.py` —— Agent 核心（完整版）

```python
from __future__ import annotations

"""
Agent 核心 —— 整合所有组件的主循环

这是整个 Agent 系统的大脑，负责：
1. 接收用户输入
2. 调用 LLM 生成响应
3. 执行工具调用
4. 管理记忆
5. 安全检查
6. 触发 Hook
"""

import asyncio
import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

from llm.base import BaseLLMClient
from tools.base import BaseTool
from memory.short_term import ShortTermMemory
from memory.long_term import LongTermMemory
from hooks.event_bus import EventBus, HookEvent, HookContext
from security.policy import SecurityPolicy, Decision
from security.auditor import Auditor
from config import AgentConfig

logger = logging.getLogger(__name__)


@dataclass
class AgentState:
    """Agent 运行时状态"""
    session_id: str = ""
    turn_count: int = 0
    is_running: bool = False
    current_task: str = ""


class Agent:
    """
    Agent 主类 —— 整合所有组件

    这是整个教程的最终产物，将前面所有章节的组件组装在一起。
    """

    def __init__(
        self,
        config: AgentConfig,
        llm_client: BaseLLMClient,
        tools: list[BaseTool] | None = None,
    ):
        self.config = config
        self.llm = llm_client
        self.tools = {t.name: t for t in (tools or [])}
        self.state = AgentState()

        # 初始化各子系统
        self.memory_short = ShortTermMemory(
            max_turns=config.memory.short_term_max_turns
        )
        self.memory_long = LongTermMemory(
            store_dir=config.resolved_data_dir / "memory"
        ) if config.memory.long_term_enabled else None

        # Hook 系统
        self.event_bus = EventBus() if config.hooks.enabled else None

        # 安全系统
        self.security = SecurityPolicy(
            strict_mode=config.security.strict_mode
        )
        self.auditor = Auditor(
            log_dir=config.security.audit_dir or None
        ) if config.security.audit_enabled else None

        # 确保数据目录存在
        config.resolved_data_dir.mkdir(parents=True, exist_ok=True)

        logger.info(f"Agent initialized: {config.name} v{config.version}")

    async def start(self, session_id: str = "default") -> None:
        """启动 Agent"""
        self.state.session_id = session_id
        self.state.is_running = True
        self.state.turn_count = 0

        # 触发启动 Hook
        await self._emit_hook(HookEvent.SESSION_START, {
            "session_id": session_id,
        })

        # 加载长期记忆
        if self.memory_long:
            await self.memory_long.load()

        logger.info(f"Agent started: session={session_id}")

    async def stop(self) -> None:
        """停止 Agent"""
        self.state.is_running = False

        # 保存记忆
        if self.memory_long:
            await self.memory_long.save()

        # 保存审计日志
        if self.auditor:
            self.auditor.save()

        # 触发停止 Hook
        await self._emit_hook(HookEvent.SESSION_END, {
            "session_id": self.state.session_id,
            "turn_count": self.state.turn_count,
        })

        logger.info(f"Agent stopped: {self.state.turn_count} turns")

    async def process_turn(self, user_input: str) -> dict[str, Any]:
        """
        处理一轮对话

        完整流程：
        1. 安全检查输入
        2. 触发前置 Hook
        3. 构建消息历史
        4. 调用 LLM
        5. 处理工具调用
        6. 安全检查输出
        7. 触发后置 Hook
        8. 更新记忆
        9. 返回结果
        """
        if not self.state.is_running:
            return {"error": "Agent 未启动"}

        self.state.turn_count += 1
        turn_id = f"turn-{self.state.turn_count}"

        logger.info(f"Turn {self.state.turn_count}: {user_input[:50]}...")

        # 1. 安全检查
        if self.security:
            filter_result = self.security.filter_output(user_input)
            # 检查输入中是否有敏感信息（可能是用户不小心粘贴的）

        # 2. 触发前置 Hook
        ctx = HookContext(
            session_id=self.state.session_id,
            turn_id=turn_id,
            data={"user_input": user_input},
        )
        await self._emit_hook(HookEvent.BEFORE_LLM, ctx)

        # 3. 构建消息历史
        messages = self.memory_short.get_messages()
        messages.append({"role": "user", "content": user_input})

        # 4. 调用 LLM
        try:
            response = await self.llm.chat(messages)
        except Exception as e:
            logger.error(f"LLM call failed: {e}")
            return {"error": f"LLM 调用失败: {e}"}

        # 5. 处理响应
        assistant_message = response.get("content", "")
        tool_calls = response.get("tool_calls", [])

        # 执行工具调用
        tool_results = []
        for tc in tool_calls:
            tool_name = tc.get("name", "")
            tool_args = tc.get("arguments", {})

            # 安全检查
            if self.security and tool_name == "bash":
                cmd = tool_args.get("command", "")
                perm = self.security.check_command(cmd)
                if perm.decision == Decision.DENY:
                    tool_results.append({
                        "tool": tool_name,
                        "error": f"命令被禁止: {perm.reason}",
                    })
                    if self.auditor:
                        self.auditor.log_command_check(
                            cmd, "denied", perm.reason, self.state.session_id
                        )
                    continue

            # 执行工具
            tool = self.tools.get(tool_name)
            if tool:
                try:
                    result = await tool.execute(**tool_args)
                    tool_results.append({
                        "tool": tool_name,
                        "result": result,
                    })
                except Exception as e:
                    tool_results.append({
                        "tool": tool_name,
                        "error": str(e),
                    })

        # 6. 安全过滤输出
        if self.security:
            assistant_message = self.security.filter_output(assistant_message)

        # 7. 更新记忆
        self.memory_short.add_message("user", user_input)
        self.memory_short.add_message("assistant", assistant_message)

        # 8. 触发后置 Hook
        ctx.data["assistant_output"] = assistant_message
        ctx.data["tool_results"] = tool_results
        await self._emit_hook(HookEvent.AFTER_LLM, ctx)

        # 9. 审计日志
        if self.auditor:
            self.auditor.log_command_check(
                user_input[:100], "processed",
                f"turn {self.state.turn_count}", self.state.session_id
            )

        return {
            "turn": self.state.turn_count,
            "content": assistant_message,
            "tool_results": tool_results,
        }

    async def _emit_hook(self, event: HookEvent, data: Any) -> None:
        """触发 Hook 事件"""
        if self.event_bus:
            ctx = data if isinstance(data, HookContext) else HookContext(
                session_id=self.state.session_id,
                turn_id=f"turn-{self.state.turn_count}",
                data=data if isinstance(data, dict) else {},
            )
            await self.event_bus.emit(event, ctx)

    def get_status(self) -> dict[str, Any]:
        """获取 Agent 状态"""
        status = {
            "name": self.config.name,
            "version": self.config.version,
            "session_id": self.state.session_id,
            "turn_count": self.state.turn_count,
            "is_running": self.state.is_running,
            "tools": list(self.tools.keys()),
            "memory_turns": self.memory_short.turn_count,
        }

        if self.security:
            status["security"] = self.security.get_security_report()

        return status
```

```python
from __future__ import annotations

"""
程序入口 —— Agent 系统的启动脚本

使用方式：
    python -m src.main              # 交互模式
    python -m src.main --config config/settings.yaml
    python -m src.main --once "帮我写一个快速排序"  # 单次执行
"""

import asyncio
import argparse
import logging
import sys
from pathlib import Path

from config import AgentConfig
from core.agent import Agent


def setup_logging(level: str = "INFO") -> None:
    """配置日志系统"""
    logging.basicConfig(
        level=getattr(logging, level.upper(), logging.INFO),
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
        datefmt="%H:%M:%S",
    )


def create_agent(config: AgentConfig) -> Agent:
    """
    创建 Agent 实例

    根据配置初始化所有依赖的组件。
    """
    # 1. 创建 LLM 客户端
    if config.llm.provider == "anthropic":
        from llm.anthropic import AnthropicClient
        llm = AnthropicClient(
            api_key=config.llm.api_key,
            model=config.llm.model,
            base_url=config.llm.base_url or None,
        )
    elif config.llm.provider == "openai":
        from llm.openai import OpenAIClient
        llm = OpenAIClient(
            api_key=config.llm.api_key,
            model=config.llm.model,
            base_url=config.llm.base_url or None,
        )
    else:
        raise ValueError(f"不支持的 LLM provider: {config.llm.provider}")

    # 2. 创建工具列表
    from tools.bash import BashTool
    from tools.filesystem import ReadFileTool, WriteFileTool, ListDirTool
    from tools.web import WebSearchTool, WebFetchTool

    tools = [
        BashTool(),
        ReadFileTool(),
        WriteFileTool(),
        ListDirTool(),
        WebSearchTool(),
        WebFetchTool(),
    ]

    # 3. 创建 Agent
    agent = Agent(
        config=config,
        llm_client=llm,
        tools=tools,
    )

    return agent


async def interactive_mode(agent: Agent) -> None:
    """
    交互模式 —— 类似 ChatGPT 的对话界面

    使用简单的 input/print 循环。
    生产环境应替换为第 12 章的 rich + prompt_toolkit CLI。
    """
    await agent.start()

    print("=" * 60)
    print(f"🦞 {agent.config.name} v{agent.config.version}")
    print(f"   模型: {agent.config.llm.model}")
    print(f"   输入 /help 查看帮助，/quit 退出")
    print("=" * 60)

    while agent.state.is_running:
        try:
            user_input = input("\n> ").strip()
        except (EOFError, KeyboardInterrupt):
            break

        if not user_input:
            continue

        # 内置命令
        if user_input == "/quit":
            break
        elif user_input == "/help":
            print("""
可用命令：
  /quit     - 退出
  /status   - 查看 Agent 状态
  /clear    - 清空短期记忆
  /help     - 显示帮助
            """)
            continue
        elif user_input == "/status":
            status = agent.get_status()
            for key, value in status.items():
                print(f"  {key}: {value}")
            continue
        elif user_input == "/clear":
            agent.memory_short.clear()
            print("✅ 短期记忆已清空")
            continue

        # 处理对话
        result = await agent.process_turn(user_input)

        if "error" in result:
            print(f"❌ 错误: {result['error']}")
        else:
            print(f"\n{result['content']}")

            # 显示工具调用结果
            for tr in result.get("tool_results", []):
                tool_name = tr.get("tool", "?")
                if "error" in tr:
                    print(f"  ⚠️  {tool_name}: {tr['error']}")
                else:
                    print(f"  ✅ {tool_name}: 执行成功")

    await agent.stop()
    print("\n👋 再见！")


async def once_mode(agent: Agent, query: str) -> None:
    """
    单次执行模式 —— 处理一个查询后退出

    适合脚本调用和自动化场景。
    """
    await agent.start()

    result = await agent.process_turn(query)

    if "error" in result:
        print(f"❌ 错误: {result['error']}", file=sys.stderr)
        sys.exit(1)
    else:
        print(result["content"])

    await agent.stop()


def parse_args() -> argparse.Namespace:
    """解析命令行参数"""
    parser = argparse.ArgumentParser(
        description="🦞 My Agent - AI 智能助手"
    )
    parser.add_argument(
        "--once", "-q",
        help="单次执行模式，处理一个查询后退出",
    )
    parser.add_argument(
        "--config", "-c",
        help="配置文件路径",
        default="config/settings.yaml",
    )
    parser.add_argument(
        "--log-level",
        choices=["DEBUG", "INFO", "WARNING", "ERROR"],
        default="INFO",
        help="日志级别",
    )
    return parser.parse_args()


def main() -> None:
    """主函数"""
    args = parse_args()

    # 配置日志
    setup_logging(args.log_level)

    # 加载配置（优先从环境变量）
    config = AgentConfig.from_env()

    # 验证配置
    errors = config.validate()
    if errors:
        print("❌ 配置验证失败：", file=sys.stderr)
        for e in errors:
            print(f"  - {e}", file=sys.stderr)
        print("\n请设置必要的环境变量：", file=sys.stderr)
        print("  export ANTHROPIC_API_KEY=sk-ant-...", file=sys.stderr)
        sys.exit(1)

    # 创建 Agent
    agent = create_agent(config)

    # 运行
    if args.once:
        asyncio.run(once_mode(agent, args.once))
    else:
        asyncio.run(interactive_mode(agent))


if __name__ == "__main__":
    main()
```

## ❌ 常见错误

### 错误 1：API Key 未设置导致启动失败

**现象**：运行 `python -m src.main` 后立即退出，提示 "LLM API Key 未设置"。

**原因**：没有设置环境变量 `ANTHROPIC_API_KEY` 或 `OPENAI_API_KEY`。

**解决**：
```bash
# 临时设置
export ANTHROPIC_API_KEY="sk-ant-api03-..."

# 永久设置（写入 ~/.bashrc 或 ~/.zshrc）
echo 'export ANTHROPIC_API_KEY="sk-ant-api03-..."' >> ~/.zshrc
source ~/.zshrc
```

### 错误 2：沙箱阻止了合法的文件操作

**现象**：Agent 无法读取 `/tmp/test.txt`，报错 "路径越界"。

**原因**：沙箱默认只允许当前工作目录，`/tmp` 不在允许范围内。

**解决**：
```bash
# 方式 1：扩大沙箱范围
export SANDBOX_ROOT=/

# 方式 2：在代码中配置多个允许根目录
from security.sandbox import SandboxConfig, FileSandbox
config = SandboxConfig(
    allowed_roots=[str(Path.cwd()), "/tmp"],
    ...
)
```

### 错误 3：记忆文件过大导致启动缓慢

**现象**：Agent 启动时需要很长时间加载记忆。

**原因**：长期记忆文件积累了大量数据，没有定期清理。

**解决**：
```bash
# 检查记忆文件大小
du -sh ~/.my_agent/memory/

# 清理过期记忆（保留最近 7 天）
find ~/.my_agent/memory/ -name "*.json" -mtime +7 -delete

# 或者在代码中设置自动清理
# 在 LongTermMemory 中添加定期清理逻辑
```

### 错误 4：Docker 容器中无法访问宿主机网络

**现象**：Agent 在 Docker 中无法调用 LLM API。

**原因**：Docker 网络配置问题，容器无法访问外部网络。

**解决**：
```yaml
# docker-compose.yml 中添加网络配置
services:
  agent:
    # ...
    extra_hosts:
      - "host.docker.internal:host-gateway"
    network_mode: bridge  # 默认就是 bridge，确保没有被覆盖
```

### 练习 2：实现配置热重载

**题目**：当 `settings.yaml` 文件变化时，自动重新加载配置（不需要重启 Agent）。

**参考答案**：
```python
import asyncio
from pathlib import Path
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class ConfigReloader(FileSystemEventHandler):
    def __init__(self, agent: Agent):
        self.agent = agent

    def on_modified(self, event):
        if event.src_path.endswith("settings.yaml"):
            # 重新加载配置
            new_config = AgentConfig.from_env()
            # 更新可变配置（如 temperature、strict_mode）

            self.agent.security.strict_mode = new_config.security.strict_mode

observer = Observer()
observer.schedule(ConfigReloader(agent), "config/", recursive=False)
observer.start()
```

### 练习 3：实现多会话管理

**题目**：让 Agent 同时支持多个独立会话，每个会话有自己的记忆和状态。

**参考答案**：
```python
class SessionManager:
    def __init__(self, config: AgentConfig):
        self.config = config
        self.sessions: dict[str, Agent] = {}

    async def create_session(self, session_id: str) -> Agent:
        agent = create_agent(self.config)
        await agent.start(session_id=session_id)
        self.sessions[session_id] = agent
        return agent

    async def get_session(self, session_id: str) -> Agent:
        if session_id not in self.sessions:
            return await self.create_session(session_id)
        return self.sessions[session_id]

    async def close_session(self, session_id: str) -> None:
        if session_id in self.sessions:
            await self.sessions[session_id].stop()
            del self.sessions[session_id]
```
