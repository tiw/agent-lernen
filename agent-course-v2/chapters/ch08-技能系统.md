## 🧠 核心概念

### 什么是技能（Skill）？

通用 Agent 什么都能做一点，但什么都不精。技能系统让 Agent 在特定领域拥有**专家级能力**——就像给一个聪明的实习生配上一本专业手册。

```
通用 Agent: "我知道 Python 的基本语法..."
+ code-review 技能: "我会检查安全漏洞、性能问题、代码风格..."
+ data-analysis 技能: "我会用 pandas 做数据清洗、可视化和统计分析..."
```

### SKILL.md 文件格式

每个技能就是一个目录 + 一个 `SKILL.md` 文件：

```
skills/
├── code-review/
│   └── SKILL.md
├── data-analysis/
│   └── SKILL.md
└── article-writing/
    └── SKILL.md
```

SKILL.md 包含 **YAML frontmatter** + **Markdown 正文**：

```markdown
---
name: code-review
description: 进行专业的代码审查
category: development
---

# 代码审查技能

你是一个资深代码审查专家。请按照以下步骤进行审查：

1. 检查安全性（SQL 注入、XSS 等）
2. 检查性能问题（N+1 查询、内存泄漏等）
3. 检查代码风格和规范
4. 提出改进建议
```

### 技能加载与激活流程

```
启动 Agent
    │
    ▼
扫描技能目录
    │
    ├── 读取每个 SKILL.md
    ├── 解析 frontmatter（YAML 元数据）
    └── 注册到技能列表
    │
    ▼
条件匹配
    │
    ├── 无条件技能 → 始终可用
    └── 条件技能 → 匹配特定场景时激活
    │
    ▼
注入系统提示
    │
    └── 将技能指令加入 Agent 的 system prompt
```

### 代码 1：`src/skills/skill.py`

```python
"""
技能数据模型。

从零手写 AI Agent 课程 · 第 8 章
"""

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Optional


@dataclass
class Skill:
    """
    技能定义。
    
    属性：
    - name: 技能名称（目录名）
    - description: 技能描述（来自 frontmatter）
    - category: 技能分类
    - content: 技能指令（Markdown 正文）
    - path: SKILL.md 文件路径
    """
    name: str
    description: str
    content: str
    path: str
    category: str = "general"
    when_to_use: str = ""
    
    def to_prompt(self) -> str:
        """将技能转换为可注入的 prompt 文本"""
        parts = [f"## 技能: {self.name}"]
        if self.description:
            parts.append(f"**描述**: {self.description}")
        if self.when_to_use:
            parts.append(f"**使用时机**: {self.when_to_use}")
        parts.append("")
        parts.append(self.content)
        return "\n".join(parts)
```

### 代码 2：`src/skills/loader.py`

```python
"""
技能加载器 —— 从目录加载 SKILL.md 文件。

从零手写 AI Agent 课程 · 第 8 章
"""

from __future__ import annotations

import os
import re
from pathlib import Path
from typing import Optional

from skills.skill import Skill


def parse_frontmatter(content: str) -> tuple[dict, str]:
    """
    解析 YAML frontmatter。
    
    格式：
    ---
    key: value
    ---
    正文内容...
    
    Returns:
        (元数据字典, 正文内容)
    """
    pattern = r'^---\s*\n(.*?)\n---\s*\n(.*)$'
    match = re.match(pattern, content, re.DOTALL)
    
    if not match:
        return {}, content.strip()
    
    yaml_text = match.group(1)
    body = match.group(2).strip()
    
    # 简单 YAML 解析（只支持 key: value 格式）
    metadata = {}
    for line in yaml_text.split('\n'):
        line = line.strip()
        if ':' in line and not line.startswith('#'):
            key, _, value = line.partition(':')
            metadata[key.strip()] = value.strip().strip('"').strip("'")
    
    return metadata, body


class SkillLoader:
    """
    技能加载器。
    
    从指定目录扫描并加载所有技能。
    
    使用示例：
        loader = SkillLoader(skills_dir="skills/")
        skills = loader.load_all()
        
        # 按名称获取
        skill = loader.get("code-review")
        
        # 按分类获取
        dev_skills = loader.get_by_category("development")
    """
    
    def __init__(self, skills_dir: str):
        """
        Args:
            skills_dir: 技能目录路径
        """
        self.skills_dir = Path(skills_dir)
        self._skills: dict[str, Skill] = {}
    
    def load_all(self) -> list[Skill]:
        """
        加载所有技能。
        
        Returns:
            技能列表
        """
        if not self.skills_dir.exists():
            return []
        
        self._skills = {}
        
        for entry in sorted(self.skills_dir.iterdir()):
            if not entry.is_dir():
                continue
            
            skill_file = entry / "SKILL.md"
            if not skill_file.exists():
                continue
            
            try:
                skill = self._load_skill(skill_file, entry.name)
                self._skills[skill.name] = skill
            except Exception as e:
                print(f"⚠️  加载技能 {entry.name} 失败: {e}")
        
        return list(self._skills.values())
    
    def _load_skill(self, file_path: Path, dir_name: str) -> Skill:
        """加载单个技能文件"""
        content = file_path.read_text(encoding='utf-8')
        metadata, body = parse_frontmatter(content)
        
        return Skill(
            name=metadata.get('name', dir_name),
            description=metadata.get('description', ''),
            content=body,
            path=str(file_path),
            category=metadata.get('category', 'general'),
            when_to_use=metadata.get('when_to_use', ''),
        )
    
    def get(self, name: str) -> Optional[Skill]:
        """按名称获取技能"""
        return self._skills.get(name)
    
    def get_by_category(self, category: str) -> list[Skill]:
        """按分类获取技能"""
        return [
            s for s in self._skills.values()
            if s.category == category
        ]
    
    def get_all_prompts(self) -> str:
        """获取所有技能的 prompt 文本（用于注入系统提示）"""
        if not self._skills:
            return ""
        
        parts = ["# 可用技能\n"]
        for skill in self._skills.values():
            parts.append(skill.to_prompt())
            parts.append("")
        
        return "\n".join(parts)
    
    @property
    def skill_names(self) -> list[str]:
        """所有技能名称"""
        return list(self._skills.keys())
    
    @property
    def count(self) -> int:
        """技能数量"""
        return len(self._skills)
```

```python
# 在 Agent.__init__ 中添加
from skills.loader import SkillLoader

# ... 在 __init__ 中 ...
# 加载技能
self.skill_loader: Optional[SkillLoader] = None
self.skills_enabled = False

def load_skills(self, skills_dir: str) -> int:
    """
    从目录加载技能。
    
    Args:
        skills_dir: 技能目录路径
        
    Returns:
        加载的技能数量
    """
    self.skill_loader = SkillLoader(skills_dir)
    skills = self.skill_loader.load_all()
    
    if skills:
        # 将技能指令注入系统提示
        skill_prompts = self.skill_loader.get_all_prompts()
        self.messages[0]["content"] += f"\n\n{skill_prompts}"
        self.skills_enabled = True
    
    return len(skills)
```

## ⚠️ 常见错误

### 错误 1：技能目录存在但没有 SKILL.md
**原因**：目录中缺少 SKILL.md 文件。
**解决**：确保每个技能目录都有 `SKILL.md` 文件。加载器会自动跳过没有 SKILL.md 的目录。

### 错误 2：frontmatter 解析失败
**原因**：YAML 格式不正确（缺少 `---` 分隔符）。
**解决**：确保 frontmatter 格式正确：
```markdown
---
key: value
---
正文
```

### 错误 3：技能指令太长，撑爆上下文
**原因**：技能太多，全部注入系统提示后超出 token 限制。
**解决**：
- 只加载需要的技能（按分类加载）
- 使用条件激活（只在需要时注入）
- 增加 `max_context_tokens`
