## 🧠 核心概念

### 为什么需要任务系统？

一个优秀的 Agent 不应该傻傻地等一个命令执行完才做下一件事。任务系统让 Agent 能：

- 🔄 **后台执行**：长时间运行的命令不阻塞对话
- 👥 **委派子任务**：把子任务交给子 Agent 处理
- ⏸️ **暂停恢复**：随时暂停和恢复任务
- 📊 **状态追踪**：实时查看任务进度

### 任务架构

```
┌─────────────────────────────────────────────┐
│              Task Registry                   │
│   tasks: { task_id → TaskState }            │
├──────────┬──────────┬───────────────────────┤
│ ShellTask│AgentTask │ MoreTask...           │
│ (后台命令)│(子Agent) │                      │
└──────────┴──────────┴───────────────────────┘
         │          │
         ▼          ▼
    subprocess   sub-agent
```

### 任务状态机

```
pending ──start──► running ──complete──► completed
                       │                     │
                    kill│                  reset│
                       ▼                     ▼
                     killed               pending
```

```python
"""
任务基类 —— 所有任务的抽象接口。

从零手写 AI Agent 课程 · 第 7 章
"""

from __future__ import annotations

import uuid
import time
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from enum import Enum
from typing import Optional


class TaskStatus(Enum):
    """任务状态"""
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    KILLED = "killed"


@dataclass
class TaskState:
    """任务状态信息"""
    id: str
    type: str
    status: TaskStatus
    description: str
    created_at: float = field(default_factory=time.time)
    updated_at: float = field(default_factory=time.time)
    output_file: Optional[str] = None
    error: Optional[str] = None
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "type": self.type,
            "status": self.status.value,
            "description": self.description,
            "created_at": self.created_at,
            "updated_at": self.updated_at,
            "output_file": self.output_file,
            "error": self.error,
        }


class Task(ABC):
    """
    任务基类。所有具体任务都必须继承此类。
    
    每个任务需要实现：
    - start(): 启动任务
    - kill(): 终止任务
    - get_output(): 获取输出
    """
    
    def __init__(self, description: str, task_id: Optional[str] = None):
        self.id = task_id or f"t_{uuid.uuid4().hex[:8]}"
        self.description = description
        self.status = TaskStatus.PENDING
        self.created_at = time.time()
        self.updated_at = time.time()
        self._output_file: Optional[str] = None
        self._error: Optional[str] = None
    
    @abstractmethod
    def start(self) -> None:
        """启动任务"""
        pass
    
    @abstractmethod
    def kill(self) -> None:
        """终止任务"""
        pass
    
    @abstractmethod
    def get_output(self, offset: int = 0) -> str:
        """
        获取任务输出。
        
        Args:
            offset: 从哪个位置开始读取（用于增量读取）
            
        Returns:
            任务输出文本
        """
        pass
    
    @property
    def is_running(self) -> bool:
        return self.status == TaskStatus.RUNNING
    
    @property
    def is_done(self) -> bool:
        return self.status in (
            TaskStatus.COMPLETED,
            TaskStatus.FAILED,
            TaskStatus.KILLED,
        )
    
    def get_state(self) -> TaskState:
        """获取任务状态"""
        return TaskState(
            id=self.id,
            type=self.__class__.__name__,
            status=self.status,
            description=self.description,
            created_at=self.created_at,
            updated_at=self.updated_at,
            output_file=self._output_file,
            error=self._error,
        )


class TaskRegistry:
    """
    任务注册表 —— 管理所有活跃任务。
    
    使用示例：
        registry = TaskRegistry()
        task = ShellTask("sleep 10", "等待 10 秒")
        registry.register(task)
        task.start()
        
        # 查看所有任务
        for state in registry.list_tasks():
            print(f"{state.id}: {state.status}")
    """
    
    def __init__(self):
        self._tasks: dict[str, Task] = {}
    
    def register(self, task: Task) -> str:
        """注册一个任务"""
        self._tasks[task.id] = task
        return task.id
    
    def get(self, task_id: str) -> Optional[Task]:
        """获取任务"""
        return self._tasks.get(task_id)
    
    def list_tasks(self) -> list[TaskState]:
        """列出所有任务的状态"""
        return [t.get_state() for t in self._tasks.values()]
    
    def kill(self, task_id: str) -> bool:
        """终止任务"""
        task = self._tasks.get(task_id)
        if task and task.is_running:
            task.kill()
            return True
        return False
    
    def cleanup_done(self) -> int:
        """清理已完成的任务"""
        done_ids = [
            tid for tid, t in self._tasks.items() if t.is_done
        ]
        for tid in done_ids:
            del self._tasks[tid]
        return len(done_ids)
```

### 代码 3：`src/tasks/shell_task.py`

```python
"""
Shell 任务 —— 在后台执行 Shell 命令。

从零手写 AI Agent 课程 · 第 7 章
"""

from __future__ import annotations

import os
import tempfile
import subprocess
import threading
import time
from typing import Optional

from tasks.base import Task, TaskStatus


class ShellTask(Task):
    """
    后台 Shell 任务。
    
    特性：
    - 异步执行，不阻塞主线程
    - 输出写入临时文件，支持增量读取
    - 支持超时自动终止
    
    使用示例：
        task = ShellTask("sleep 5 && echo done", "等待 5 秒")
        task.start()
        # ... 做其他事 ...
        output = task.get_output()
    """
    
    def __init__(
        self,
        command: str,
        description: str,
        timeout: int = 300,
        task_id: Optional[str] = None,
        cwd: Optional[str] = None,
    ):
        super().__init__(description, task_id)
        self.command = command
        self.timeout = timeout
        self.cwd = cwd or os.getcwd()
        self._process: Optional[subprocess.Popen] = None
        self._thread: Optional[threading.Thread] = None
        self._output_file: Optional[str] = None
    
    def start(self) -> None:
        """启动后台命令"""
        if self.status != TaskStatus.PENDING:
            raise RuntimeError(f"Task {self.id} is not in PENDING state")
        
        # 创建临时输出文件
        self._output_file = tempfile.mktemp(
            prefix=f"shell_{self.id}_",
            suffix=".log",
        )
        
        self.status = TaskStatus.RUNNING
        self.updated_at = time.time()
        
        # 在后台线程中执行
        self._thread = threading.Thread(
            target=self._run,
            daemon=True,
        )
        self._thread.start()
    
    def _run(self) -> None:
        """在后台线程中执行命令"""
        try:
            with open(self._output_file, 'w') as out_f:
                self._process = subprocess.Popen(
                    self.command,
                    shell=True,
                    stdout=out_f,
                    stderr=subprocess.STDOUT,
                    cwd=self.cwd,
                    text=True,
                )
                
                try:
                    self._process.wait(timeout=self.timeout)
                except subprocess.TimeoutExpired:
                    self._process.kill()
                    out_f.write(f"\n[超时] 命令执行超过 {self.timeout} 秒\n")
                
                if self._process.returncode == 0:
                    self.status = TaskStatus.COMPLETED
                else:
                    self.status = TaskStatus.FAILED
                    self._error = f"Exit code: {self._process.returncode}"
        
        except Exception as e:
            self.status = TaskStatus.FAILED
            self._error = str(e)
        
        finally:
            self.updated_at = time.time()
    
    def kill(self) -> None:
        """终止任务"""
        if self._process and self._process.poll() is None:
            self._process.kill()
        self.status = TaskStatus.KILLED
        self.updated_at = time.time()
    
    def get_output(self, offset: int = 0) -> str:
        """获取任务输出（支持增量读取）"""
        if not self._output_file or not os.path.exists(self._output_file):
            return ""
        
        with open(self._output_file, 'r') as f:
            f.seek(offset)
            return f.read()
    
    @property
    def output_size(self) -> int:
        """获取输出文件大小"""
        if not self._output_file or not os.path.exists(self._output_file):
            return 0
        return os.path.getsize(self._output_file)
```

```python
"""
Agent 子任务 —— 委派给子 Agent 执行。

从零手写 AI Agent 课程 · 第 7 章
"""

from __future__ import annotations

import tempfile
import threading
import time
from typing import Optional, TYPE_CHECKING

from tasks.base import Task, TaskStatus

if TYPE_CHECKING:
    from agent import Agent


class AgentTask(Task):
    """
    Agent 子任务。
    
    将子任务委派给一个独立的 Agent 实例执行。
    
    特性：
    - 独立的 Agent 实例（有自己的消息历史和工具）
    - 异步执行，不阻塞主线程
    - 输出写入文件，支持增量读取
    
    使用示例：
        sub_agent = Agent(system_prompt="你是一个代码审查专家")
        task = AgentTask(
            sub_agent=sub_agent,
            user_input="审查以下代码的安全性：...",
            description="代码安全审查"
        )
        task.start()
    """
    
    def __init__(
        self,
        sub_agent: 'Agent',
        user_input: str,
        description: str,
        task_id: Optional[str] = None,
    ):
        super().__init__(description, task_id)
        self.sub_agent = sub_agent
        self.user_input = user_input
        self._result: Optional[str] = None
        self._output_file: Optional[str] = None
        self._thread: Optional[threading.Thread] = None
    
    def start(self) -> None:
        """启动子任务"""
        if self.status != TaskStatus.PENDING:
            raise RuntimeError(f"Task {self.id} is not in PENDING state")
        
        self._output_file = tempfile.mktemp(
            prefix=f"agent_{self.id}_",
            suffix=".log",
        )
        
        self.status = TaskStatus.RUNNING
        self.updated_at = time.time()
        
        self._thread = threading.Thread(
            target=self._run,
            daemon=True,
        )
        self._thread.start()
    
    def _run(self) -> None:
        """执行子任务"""
        try:
            # 调用子 Agent
            result = self.sub_agent.chat(self.user_input)
            self._result = result
            
            # 写入输出文件
            with open(self._output_file, 'w') as f:
                f.write(result)
            
            self.status = TaskStatus.COMPLETED
        
        except Exception as e:
            self.status = TaskStatus.FAILED
            self._error = str(e)
            
            with open(self._output_file, 'w') as f:
                f.write(f"[错误] {e}")
        
        finally:
            self.updated_at = time.time()
    
    def kill(self) -> None:
        """终止子任务"""
        # Agent 任务无法强制终止，只能标记
        self.status = TaskStatus.KILLED
        self.updated_at = time.time()
    
    def get_output(self, offset: int = 0) -> str:
        """获取任务输出"""
        if self._result:
            return self._result[offset:]
        
        if self._output_file:
            try:
                with open(self._output_file, 'r') as f:
                    f.seek(offset)
                    return f.read()
            except FileNotFoundError:
                pass
        
        return ""
```

```python
    """获取任务输出"""
    task = self.task_registry.get(task_id)

        return f"[错误] 任务 {task_id} 不存在"
    return task.get_output(offset)

    """终止任务"""
```

```python
"""测试任务系统"""
import time
from tasks.base import TaskRegistry
from tasks.shell_task import ShellTask


def test_shell_task():
    """测试 Shell 任务"""
    # 快速任务
    task = ShellTask("echo 'Hello from background!'", "快速测试")
    task.start()
    time.sleep(1)  # 等待完成
    
    output = task.get_output()
    print(f"✅ 快速任务输出: {output.strip()}")
    assert task.status.value == "completed"
    
    # 长时间任务
    task2 = ShellTask("sleep 3 && echo 'Done after 3s'", "等待 3 秒")
    task2.start()
    
    # 立即检查（应该还在运行）
    assert task2.is_running
    print("✅ 任务正在运行")
    
    # 增量读取
    output1 = task2.get_output()
    time.sleep(1)
    output2 = task2.get_output(offset=len(output1))
    print(f"✅ 增量读取: 新增 {len(output2)} 字符")
    
    # 等待完成
    time.sleep(3)
    assert task2.is_done
    print(f"✅ 任务完成: {task2.get_output().strip()}")


def test_task_registry():
    """测试任务注册表"""
    registry = TaskRegistry()
    
    # 注册多个任务
    for i in range(3):
        task = ShellTask(f"echo 'Task {i}'", f"任务 {i}")
        registry.register(task)
        task.start()

    # 列出所有任务
    states = registry.list_tasks()
    print(f"✅ 活跃任务数: {len(states)}")
```

## ⚠️ 常见错误

### 错误 1：后台任务输出为空
**原因**：任务还没执行完就读取输出。
**解决**：等待任务完成或检查 `task.is_running`。

### 错误 2：线程泄漏
**原因**：大量创建任务但没有等待完成，线程堆积。
**解决**：使用 `daemon=True`（已设置），或定期调用 `registry.cleanup_done()`。

### 错误 3：临时文件未清理
**原因**：任务的输出文件在任务完成后仍然存在。
**解决**：添加清理逻辑：
```python
import os
if task._output_file and os.path.exists(task._output_file):
    os.unlink(task._output_file)
```
