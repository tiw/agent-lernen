## 🧠 核心概念

### 为什么需要多智能体？

一个 Agent 再聪明也有极限。当任务复杂到需要分工协作时，就需要**多智能体系统**：

```
单 Agent: 用户 → Agent → 结果
多 Agent: 用户 → 协调器 → [研究者, 编码者, 审查者] → 结果
```

### 多智能体架构

```
┌─────────────────────────────────────────────────┐
│                 协调器 (Coordinator)              │
│   接收用户任务 → 分解 → 分配 → 汇总 → 回复        │
├─────────────────────────────────────────────────┤
│                  消息总线 (Bus)                   │
│   所有 Agent 通过总线交换消息                     │
├──────────┬──────────┬──────────┬────────────────┤
│ 研究者    │ 编码者    │ 审查者    │ ...            │
│ (Research)│ (Coder)  │ (Reviewer)│                │
└──────────┴──────────┴──────────┴────────────────┘
```

### 核心组件

| 组件 | 作用 | 类比 |
|------|------|------|
| **角色定义** | 每个 Agent 的职责和工具集 | 团队成员的岗位描述 |
| **消息总线** | Agent 间的通信通道 | 团队群聊 |
| **任务板** | 任务的创建、分配、跟踪 | 看板（Trello/Jira） |
| **协调器** | 分解任务、分配工作、汇总结果 | 项目经理 |

```python
"""
消息总线 —— Agent 间的通信通道。

从零手写 AI Agent 课程 · 第 10 章
"""

from __future__ import annotations

import time
import uuid
from dataclasses import dataclass, field
from typing import Optional


@dataclass
class BusMessage:
    """总线消息"""
    id: str
    sender: str           # 发送者角色
    recipient: str        # 接收者角色（"all" 表示广播）
    content: str          # 消息内容
    timestamp: float = field(default_factory=time.time)
    reply_to: Optional[str] = None  # 回复的消息 ID
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "sender": self.sender,
            "recipient": self.recipient,
            "content": self.content,
            "timestamp": self.timestamp,
        }


class MessageBus:
    """
    消息总线。
    
    所有 Agent 通过总线交换消息。
    支持点对点消息和广播。
    
    使用示例：
        bus = MessageBus()
        bus.send("researcher", "coder", "我找到了相关文档...")
        bus.broadcast("coordinator", "所有角色，开始工作！")
        
        # 获取某个角色的消息
        messages = bus.get_messages_for("coder")
    """
    
    def __init__(self, max_messages: int = 1000):
        """
        Args:
            max_messages: 最大消息数（防止内存泄漏）
        """
        self.max_messages = max_messages
        self._messages: list[BusMessage] = []
    
    def send(
        self,
        sender: str,
        recipient: str,
        content: str,
        reply_to: Optional[str] = None,
    ) -> str:
        """
        发送点对点消息。
        
        Returns:
            消息 ID
        """
        msg = BusMessage(
            id=str(uuid.uuid4())[:8],
            sender=sender,
            recipient=recipient,
            content=content,
            reply_to=reply_to,
        )
        self._messages.append(msg)
        
        # 清理旧消息
        if len(self._messages) > self.max_messages:
            self._messages = self._messages[-self.max_messages:]
        
        return msg.id
    
    def broadcast(self, sender: str, content: str) -> str:
        """
        广播消息（所有角色都能收到）。
        
        Returns:
            消息 ID
        """
        return self.send(sender, "all", content)
    
    def get_messages_for(self, role: str) -> list[BusMessage]:
        """获取指定角色的未读消息"""
        return [
            msg for msg in self._messages
            if msg.recipient == role or msg.recipient == "all"
        ]
    
    def get_all_messages(self) -> list[BusMessage]:
        """获取所有消息"""
        return list(self._messages)
    
    def clear(self):
        """清空消息"""
        self._messages.clear()
    
    @property
    def message_count(self) -> int:
        return len(self._messages)
```

```python
"""
任务板 —— 多智能体任务的创建、分配和跟踪。

从零手写 AI Agent 课程 · 第 10 章
"""

from __future__ import annotations

import time
import uuid
from dataclasses import dataclass, field
from enum import Enum
from typing import Optional


class TaskStatus(Enum):
    """任务状态"""
    TODO = "todo"
    IN_PROGRESS = "in_progress"
    DONE = "done"
    FAILED = "failed"


@dataclass
class TeamTask:
    """团队任务"""
    id: str
    title: str
    description: str
    assignee: str       # 负责的角色
    status: TaskStatus = TaskStatus.TODO
    result: str = ""
    created_at: float = field(default_factory=time.time)
    updated_at: float = field(default_factory=time.time)
    error: str = ""
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "title": self.title,
            "description": self.description,
            "assignee": self.assignee,
            "status": self.status.value,
            "result": self.result[:200] if self.result else "",
        }


class TaskBoard:
    """
    任务板。
    
    管理多智能体协作中的任务生命周期。
    
    使用示例：
        board = TaskBoard()
        task = board.create("researcher", "调研 Python 3.13 新特性")
        board.start(task.id)
        board.complete(task.id, "调研报告：...")
    """
    
    def __init__(self):
        self._tasks: dict[str, TeamTask] = {}
    
    def create(
        self,
        assignee: str,
        title: str,
        description: str = "",
    ) -> TeamTask:
        """创建任务"""
        task = TeamTask(
            id=str(uuid.uuid4())[:8],
            title=title,
            description=description or title,
            assignee=assignee,
        )
        self._tasks[task.id] = task
        return task
    
    def start(self, task_id: str) -> bool:
        """开始任务"""
        task = self._tasks.get(task_id)
        if task and task.status == TaskStatus.TODO:
            task.status = TaskStatus.IN_PROGRESS
            task.updated_at = time.time()
            return True
        return False
    
    def complete(self, task_id: str, result: str) -> bool:
        """完成任务"""
        task = self._tasks.get(task_id)
        if task and task.status == TaskStatus.IN_PROGRESS:
            task.status = TaskStatus.DONE
            task.result = result
            task.updated_at = time.time()
            return True
        return False
    
    def fail(self, task_id: str, error: str) -> bool:
        """标记任务失败"""
        task = self._tasks.get(task_id)
        if task:
            task.status = TaskStatus.FAILED
            task.error = error
            task.updated_at = time.time()
            return True
        return False
    
    def get(self, task_id: str) -> Optional[TeamTask]:
        return self._tasks.get(task_id)
    
    def get_pending(self, assignee: Optional[str] = None) -> list[TeamTask]:
        """获取待处理的任务"""
        tasks = [
            t for t in self._tasks.values()
            if t.status in (TaskStatus.TODO, TaskStatus.IN_PROGRESS)
        ]
        if assignee:
            tasks = [t for t in tasks if t.assignee == assignee]
        return tasks
    
    def get_done(self) -> list[TeamTask]:
        """获取已完成的任务"""
        return [
            t for t in self._tasks.values()
            if t.status == TaskStatus.DONE
        ]
    
    def summary(self) -> str:
        """获取任务板摘要"""
        lines = ["📋 任务板："]
        for task in self._tasks.values():
            status_emoji = {
                TaskStatus.TODO: "⏳",
                TaskStatus.IN_PROGRESS: "🔄",
                TaskStatus.DONE: "✅",
                TaskStatus.FAILED: "❌",
            }.get(task.status, "❓")
            lines.append(
                f"  {status_emoji} [{task.assignee}] {task.title}"
            )
        return "\n".join(lines)
```

```python
"""
协调器 —— 多智能体协作的指挥中心。

从零手写 AI Agent 课程 · 第 10 章
"""

from __future__ import annotations

from typing import Optional, TYPE_CHECKING

from multi_agent.message_bus import MessageBus
from multi_agent.task_board import TaskBoard
from multi_agent.role import Role

if TYPE_CHECKING:
    from agent import Agent


class Coordinator:
    """
    多智能体协调器。
    
    职责：
    1. 接收用户任务
    2. 分解为子任务
    3. 分配给不同角色
    4. 跟踪进度
    5. 汇总结果
    
    使用示例：
        coordinator = Coordinator()
        coordinator.add_role(researcher_role)
        coordinator.add_role(coder_role)
        
        result = coordinator.run(
            user_task="调研 FastAPI 并创建一个简单的 API 服务",
            max_rounds=5
        )
    """
    
    def __init__(self, model: str = "gpt-4o-mini"):
        self.model = model
        self.bus = MessageBus()
        self.board = TaskBoard()
        self._roles: dict[str, Role] = {}
        self._agents: dict[str, 'Agent'] = {}
    
    def add_role(self, role: Role) -> None:
        """添加一个角色"""
        self._roles[role.name] = role
        self._agents[role.name] = role.create_agent(model=self.model)
    
    def run(
        self,
        user_task: str,
        max_rounds: int = 5,
    ) -> str:
        """
        执行多智能体协作。
        
        Args:
            user_task: 用户任务描述
            max_rounds: 最大协作轮数
            
        Returns:
            最终结果
        """
        if not self._roles:
            return "❌ 没有可用的角色。请先添加角色。"
        
        # 广播任务
        self.bus.broadcast("coordinator", f"新任务: {user_task}")
        
        # 简化版：每个角色依次执行
        results = []
        
        for role_name, agent in self._agents.items():
            # 创建任务
            task = self.board.create(
                assignee=role_name,
                title=f"{role_name} 处理: {user_task[:50]}",
            )
            self.board.start(task.id)
            
            # 收集上下文（其他角色的结果）
            context = ""
            if results:
                context = "\n\n其他角色的结果：\n"
                for r_name, r_result in results:
                    context += f"[{r_name}]: {r_result[:500]}\n"
            
            # 执行
            prompt = f"任务: {user_task}\n{context}\n\n请完成你的部分。"
            
            try:
                result = agent.chat(prompt)
                self.board.complete(task.id, result)
                results.append((role_name, result))
                
                # 通知其他角色
                self.bus.send(
                    role_name, "all",
                    f"[{role_name}] 完成: {result[:200]}..."
                )
            except Exception as e:
                self.board.fail(task.id, str(e))
                results.append((role_name, f"[错误] {e}"))
        
        # 汇总
        summary = ["📊 多智能体协作结果：\n"]
        summary.append(self.board.summary())
        summary.append("")
        
        for role_name, result in results:
            summary.append(f"--- {role_name} ---")
            summary.append(result[:1000])
            summary.append("")
        
        return "\n".join(summary)
    
    def get_status(self) -> dict:
        """获取协作状态"""
        return {
            "roles": list(self._roles.keys()),
            "tasks": [t.to_dict() for t in self.board._tasks.values()],
            "messages": self.bus.message_count,
        }
```

## ⚠️ 常见错误

### 错误 1：角色间消息丢失
**原因**：`get_messages_for()` 返回所有消息（包括已读的）。
**解决**：添加已读标记，或在获取后清理。

### 错误 2：协调器无限循环
**原因**：角色间互相依赖，形成循环等待。
**解决**：设置 `max_rounds` 上限，超时后强制汇总。

### 错误 3：子 Agent 消耗过多 Token
**原因**：每个角色都是独立的 Agent，有自己的消息历史。
**解决**：限制每个角色的 `max_turns`，或定期 `reset()`。
